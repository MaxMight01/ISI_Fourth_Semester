\chapter{MODULES}

\section{Abstracting Vector Spaces}

Just as in the way that vector spaces are defined over fields, we can analogously define modules over rings. The theory of modules extends beyond commutative rings with unity to non-commutative rings as well, but we will restrict ourselves to the commutative case.

\begin{definition}
    Let $R$ be a ring. An $R$-\eax{module} $(M,+,\cdot,R)$ where $+ : M \times M \to M$ is a binary operation and $\cdot : R \times M \to M$ is a function satisfying the following axioms.
    \begin{itemize}
        \item $(M,+)$ is an abelian group.
        \item $r \cdot (m_{1} + m_{2}) = r \cdot m_{1} + r \cdot m_{2}$ for all $r \in R$ and $m_{1},m_{2} \in M$.
        \item $(r_{1} + r_{2}) \cdot m = r_{1} \cdot m + r_{2} \cdot m$ for all $r_{1},r_{2} \in R$ and $m \in M$.
        \item $1 \cdot m = m$ for all $m \in M$.
        \item $(r_{1}r_{2}) \cdot m = r_{1} \cdot (r_{2} \cdot m)$ for all $r_{1},r_{2} \in R$ and $m \in M$.
    \end{itemize}
\end{definition}

Note that if $R$ is a field, then an $R$-module is precisely a vector space over $R$.

\begin{example}
    As a non-trivial example of a module, let $R = \Z$. Then a $\Z$-module is precisely an abelian group, since the axioms of a $\Z$-module reduce to the axioms of an abelian group; one can show this. Conversely, any abelian group can be regarded as a $\Z$-module by defining $n \cdot m = m + m + \cdots + m$ ($n$ times) for $n > 0$, $0 \cdot m = 0$ for all $m \in M$, and $(-n) \cdot m = -(n \cdot m)$ for $n > 0$. More trivially, if $R$ is a ring, then $R$ itself is an $R$-module by the natural mappings. If $I$ is an ideal of $R$, then $I$ is an $R$-module by the natural mappings.
\end{example}

\begin{example}
    Suppose $\varphi:R \to R'$ is a homomorphism of rings. Then $R'$ is an $R$-module by defining $r \cdot r' = \varphi(r)r'$ for all $r \in R$ and $r' \in R'$. In particular, if $\varphi$ is an inclusion map, then $R'$ is an $R$-module by the natural mappings. More particularly, $S^{-1}R$ is an $R$-module for any multiplicatively closed subset $S$ of $R$ by the natural mappings.
\end{example}

Some trivial consequences of the axioms of a module can be inferred easily. For instance, $0 \cdot m = 0$ for all $m \in M$ since $0 \cdot m = (0 + 0) \cdot m = 0 \cdot m + 0 \cdot m$ implies $0 \cdot m = 0$. Similarly, $r \cdot 0 = 0$ for all $r \in R$. Also, $(-1) \cdot m = -m$ for all $m \in M$ since $0 = 0 \cdot m = (1 + (-1)) \cdot m = 1 \cdot m + (-1) \cdot m = m + (-1) \cdot m$ implies $(-1) \cdot m = -m$. More generally, $(-r) \cdot m = -(r \cdot m)$ for all $r \in R$ and $m \in M$.

\begin{definition}
    Let $M$ be an $R$-module. A non-empty subset $N \subseteq M$ is called an $R$-\eax{submodule} of $M$ if for all $x,y \in N$ and $r \in R$, $rx + y \in N$. That is, $(N, +|_{N \times N}, \cdot|_{R \times N}, R)$ is an $R$-module.
\end{definition}

From the above, any ideal $I$ of $R$ is a submodule of the $R$-module $R$.

\begin{proposition}
    Let $R$ be a ring and $M$ be an $R$-module. Also let $N$ be a submodule of $M$. Then $M/N$ has a natural structure of an $R$-module defined by $R \times M/N \to M/N$ given by $(r,m+N) \mapsto r \cdot (m+N) \defeq rm+N$ for all $r \in R$ and $m \in M$.
\end{proposition}

This is called the \eax{quotient module} of $M$ by $N$.

\begin{proof}
    First, we verify that the above definition is well-defined. Let $m + N = m' + N$ for some $m,m' \in M$. Then $m - m' \in N$, so $r(m - m') \in N$ since $N$ is a submodule of $M$. Hence, $rm + N = rm' + N$, so the above definition is well-defined. It is left as an exercise to verify that the above definition indeed gives $M/N$ the structure of an $R$-module, by checking the axioms of an $R$-module.
\end{proof}

\subsection{Linear Maps, and Isomorphisms}

\begin{definition}
    Let $M,N$ be $R$-modules. A map $\varphi:M \to N$ is called $R$-\eax{linear}, or $R$-\eax{module homomorphism}, if $\varphi(x+y) = \varphi(x) + \varphi(y)$ for all $x,y \in M$ and $\varphi(rx) = r\varphi(x)$ for all $r \in R$ and $x \in M$.
\end{definition}

Equivalently, $\varphi$ is $R$-linear if and only if $\varphi(rx+y) = r\varphi(x) + \varphi(y)$ for all $r \in R$ and $x,y \in M$.

\begin{example}
    If $N \subseteq M$ is an $R$-submodule, then the inclusion map $i:N \to M$ is $R$-linear, and injective. Also, the quotient map $q:M \to M/N$ given by $m \mapsto m + N$ is $R$-linear, and surjective. 
\end{example}

An $R$-\eax{module isomorphism} is simply a bijective $R$-linear map. If there exists an $R$-module isomorphism between $M$ and $N$, then we say that $M$ and $N$ are isomorphic as $R$-modules.

\begin{proposition}
    Let $\varphi : M \to N$ be an $R$-linear map. Then $\ker \varphi = \{m \in M \mid \varphi(m) = 0\}$ is a submodule of $M$, and $\Img \varphi = \{\varphi(m) \mid m \in M\}$ is a submodule of $N$. M
\end{proposition}
\begin{proof}
    Let $x,y \in \ker \varphi$ and $r \in R$. Then $\varphi(rx+y) = r\varphi(x) + \varphi(y) = 0$, so $rx+y \in \ker \varphi$. Hence, $\ker \varphi$ is a submodule of $M$. Let $x,y \in \Img \varphi$ and $r \in R$. Then there exist $m_{1},m_{2} \in M$ such that $\varphi(m_{1}) = x$ and $\varphi(m_{2}) = y$. Then $\varphi(rm_{1}+m_{2}) = r\varphi(m_{1}) + \varphi(m_{2}) = rx + y$, so $rx+y \in \Img \varphi$. Hence, $\Img \varphi$ is a submodule of $N$.
\end{proof}

\begin{theorem}[The first isomorphism theorem for modules]
    Let $\varphi : M \to N$ be an $R$-linear map. Let $K = \ker \varphi$ and $K' \subseteq K$ be an $R$-submodule. Then $\varphi$ induces an $R$-linear map $\overline{\varphi} : M/K' \to N$ such that $\overline{\varphi}(m+K') = \varphi(m)$ for all $m \in M$. Moreover, $\overline{\varphi} \circ q = \varphi$, where $q : M \to M/K'$ is the quotient map. In particular, if $K' = K$, then $\overline{\varphi} : M/K \to N$ is an injective $R$-linear map, and if $\varphi$ is surjective, then $\overline{\varphi} : M/K \to N$ is an $R$-module isomorphism.
\end{theorem}

\begin{proof}
    The map $\overline{\varphi}$ is well-defined since if $m + K' = m' + K'$ for some $m,m' \in M$, then $m - m' \in K' \subseteq K$, so $\varphi(m) = \varphi(m')$. Let $m+K',n+K' \in M/K'$ and $r \in R$. Then $\overline{\varphi}((m+K')+(n+K')) = \overline{\varphi}((m+n)+K') = \varphi(m+n) = \varphi(m) + \varphi(n) = \overline{\varphi}(m+K') + \overline{\varphi}(n+K')$ and $\overline{\varphi}(r(m+K')) = \overline{\varphi}(rm+K') = \varphi(rm) = r\varphi(m) = r\overline{\varphi}(m+K')$, so $\overline{\varphi}$ is $R$-linear. Also, for $m \in M$, $\overline{\varphi}(q(m)) = \overline{\varphi}(m+K') = \varphi(m)$, so $\overline{\varphi} \circ q = \varphi$. If $K' = K$, then $\ker \overline{\varphi} = \{m + K \mid m \in M, \varphi(m) = 0\} = \{m+K \mid \varphi(m) = 0\} = \{0+K\}$, so $\overline{\varphi}$ is injective. If $\varphi$ is surjective, then $\overline{\varphi} \circ q = \varphi$ is surjective, so $\overline{\varphi}$ is surjective. Hence, $\overline{\varphi}$ is an $R$-module isomorphism.
\end{proof}

\begin{theorem}[The second isomorphism theorem for modules]
    Let $M$ be an $R$-module, and let $N_{1},N_{2}$ be $R$-submodules of $M$. Then $N_{1} + N_{2} = \{n_{1} + n_{2} \mid n_{1} \in N_{1}, n_{2} \in N_{2}\}$ is an $R$-submodule of $M$, and $N_{1} \cap N_{2}$ is an $R$-submodule of $M$. Moreover,
    \begin{align}
        (N_{1}+N_{2})/N_{2} \cong N_{1}/(N_{1} \cap N_{2})
    \end{align}
\end{theorem}

\begin{proof}
    Let $n_{1},n_{1}' \in N_{1}$ and $n_{2},n_{2}' \in N_{2}$, and let $r \in R$. Then $(n_{1}+n_{2}) + (n_{1}'+n_{2}') = (n_{1}+n_{1}') + (n_{2}+n_{2}') \in N_{1} + N_{2}$ and $r(n_{1}+n_{2}) = rn_{1} + rn_{2} \in N_{1} + N_{2}$, so $N_{1} + N_{2}$ is an $R$-submodule of $M$. Let $x,y \in N_{1} \cap N_{2}$ and $r \in R$. Then $x+y \in N_{1}$ and $x+y \in N_{2}$, so $x+y \in N_{1} \cap N_{2}$. Also, $rx \in N_{1}$ and $rx \in N_{2}$, so $rx \in N_{1} \cap N_{2}$. Hence, $N_{1} \cap N_{2}$ is an $R$-submodule of $M$. Define $\varphi : N_{1} \to (N_{1}+N_{2})/N_{2}$ by $\varphi(n) = n + N_{2}$ for all $n \in N_{1}$. Then $\varphi$ is well-defined since if $n,n' \in N_{1}$ and $\varphi(n) = \varphi(n')$, then $n - n' \in N_{2}$, so $n - n' \in N_{1} \cap N_{2}$. Also, for all $n,n' \in N$ and $r \in R$, $\varphi(n+n') = (n+n') + N_{2} = (n+N_{2}) + (n'+N_{2}) = \varphi(n) + \varphi(n')$ and $\varphi(rn) = rn + N_{2} = r(n+N_{2}) = r\varphi(n)$, so $\varphi$ is an $R$-linear map. Moreover, $\ker \varphi = N_{1} \cap N_{2}$ since if $n \in N_{1} \cap N_{2}$, then $\varphi(n) = n + N_{2} = N_{2}$, so $n \in \ker \varphi$, and if $n \in \ker \varphi$, then $\varphi(n) = n + N_{2} = N_{2}$, so $n \in N_{2}$, and since $n \in N_{1}$, $n \in N_{1} \cap N_{2}$. Hence, $\ker \varphi = N_{1} \cap N_{2}$. Also, $\Img \varphi = \{n + N_{2} \mid n \in N_{1}\} = (N_{1}+N_{2})/N_{2}$ since if $n \in N_{1}$, then $\varphi(n) = n + N_{2} \in (N_{1}+N_{2})/N_{2}$, and if $n + N_{2} \in (N_{1}+N_{2})/N_{2}$, then there exist $n_{1} \in N_{1}$ and $n_{2} \in N_{2}$ such that $n + N_{2} = (n_{1}+n_{2}) + N_{2} = n_{1} + N_{2}$, so $n_{1} \in N_{1}$ and $\varphi(n_{1}) = n_{1} + N_{2} = n + N_{2}$, so $n + N_{2} \in \Img \varphi$. Hence, $\Img \varphi = (N_{1}+N_{2})/N_{2}$. By the first isomorphism theorem for modules, $N_{1}/(N_{1} \cap N_{2}) \cong (N_{1}+N_{2})/N_{2}$.
\end{proof}


\begin{theorem}[The third isomorphism theorem for modules]
    Let $M$ be an $R$-module, and let $K \subseteq N \subseteq M$ be $R$-submodules. Then $N/K$ is an $R$-submodule of $M/K$, and $(M/K)/(N/K) \cong M/N$.    
\end{theorem}
The proof is left as an exercise, by defining $\varphi : M/K \to M/N$ by $\varphi(m+K) = m + N$ for all $m \in M$.

\begin{proposition}
    Let $M$ and $N$ be $R$-modules and $\varphi,\psi: M \to N$ be $R$-linear maps. Then $\varphi + \psi : M \to N$ defined by $(\varphi + \psi)(m) = \varphi(m) + \psi(m)$ for all $m \in M$ is an $R$-linear map, and $r\varphi : M \to N$ defined by $(r\varphi)(m) = r\varphi(m)$ for all $m \in M$ is an $R$-linear map for all $r \in R$. 
\end{proposition}

\begin{proof}
    Let $m,m' \in M$ and $s \in R$. Then $(\varphi + \psi)(sm+m') = \varphi(sm+m') + \psi(sm+m') = \varphi(sm) + \varphi(m') + \psi(sm) + \psi(m') = (s\varphi(m) + s\psi(m)) + (\varphi(m') + \psi(m')) = s(\varphi + \psi)(m) + (\varphi + \psi)(m')$, and $(r\varphi)(sm+m') = r\varphi(sm+m') = r(s\varphi(m) + \varphi(m')) = s(r\varphi)(m) + (r\varphi)(m')$, so $\varphi + \psi$ and $r\varphi$ are $R$-linear maps.
\end{proof}

Thus, if we look at all $R$-linear maps from $M$ to $N$, we in turn get another $R$-module.

\begin{proposition}
    Let $M$ and $N$ be $R$-modules. The set of all $R$-linear maps from $M$ to $N$, denoted by $\Hom_{R}(M,N)$, is an $R$-module under the operations defined in the previous proposition.
\end{proposition}

This proof is left as an exercise, by verifying the axioms of an $R$-module for $\Hom_{R}(M,N)$ under the operations defined in the previous proposition. We also define $\End_{R}(M) = \Hom_{R}(M,M)$, which is the set of all $R$-linear maps from $M$ to itself. Note that $\End_{R}(M)$ is not just an $R$-module, but also a (possibly non-commutative) ring under composition of maps, with the identity map as the multiplicative identity. One can show also show that $\Hom_{R}(R,M) \cong M$ as $R$-modules, by defining $\varphi : \Hom_{R}(R,M) \to M$ by $\varphi(f) = f(1)$ for all $f \in \Hom_{R}(R,M)$, and showing that $\varphi$ is an $R$-module isomorphism. Also, if $M,N$ are $R$-modules, then $M \oplus N$ is an $R$-module under the natural operations.

\begin{definition}
    Let $M$ be a $R$-module, and $S \subseteq M$ be a subset. Then the submodule of $M$ generated by $S$, denoted by $\ip{S}$, is the smallest submodule of $M$ containing $S$. We call $S$ a generating set for $M$ if $\ip{S} = M$. $S$ is called \eax{linearly independent} if for all $n \in \N$, $r_{1},\ldots,r_{n} \in R$ and $s_{1},\ldots,s_{n} \in S$, $\sum_{i=1}^{n} r_{i}s_{i} = 0$ implies $r_{1}s_{1} = r_{2}s_{2} = \cdots = r_{n}s_{n} = 0$. A linearly independent generating set for $M$ is called a \eax{basis} for $M$.
\end{definition}

\begin{proposition}
    For a subset $S$ of an $R$-module $M$, $\ip{S} = \{\sum_{i=1}^{n} r_{i}s_{i} \mid n \in \N, r_{1},\ldots,r_{n} \in R, s_{1},\ldots,s_{n} \in S\}$.
\end{proposition}

\begin{definition}
    An $R$-module $M$ is called a \eax{free module} if $M$ has a basis.
\end{definition}

For example, if $R$ is a ring, then $R^{n}$ is a free $R$-module for all $n \in \N$, with the standard basis $e_{1},\ldots,e_{n}$.

\begin{proposition}
    A finitely generated free $R$-module $M$ is free if and only if $M \cong R^{n}$ for some $n \in \N$.
\end{proposition}
\begin{proof}
    If $M \cong \R^{n}$ via linear isomorphism $\varphi : \R^{n} \to M$, then set $x_{i} \defeq \varphi(e_{i})$ for all $1 \leq i \leq n$. If $\sum_{i=1}^{n} r_{i}x_{i} = 0$ for some $r_{1},\ldots,r_{n} \in R$, then $\sum_{i=1}^{n} r_{i}e_{i} = \varphi^{-1}(\sum_{i=1}^{n} r_{i}x_{i}) = \sum_{i=1}^{n} r_{i}\varphi^{-1}(x_{i}) = \varphi^{-1}(0) = 0$, so $\sum_{i=1}^{n} r_{i}e_{i} = 0$, so $r_{1}e_{1} = r_{2}e_{2} = \cdots = r_{n}e_{n} = 0$, so $r_{1}x_{1} = r_{2}x_{2} = \cdots = r_{n}x_{n} = 0$. Also, for $m \in M$, there exist $r_{1},\ldots,r_{n} \in R$ such that $\varphi(\sum_{i=1}^{n} r_{i}e_{i}) = m$, so $\sum_{i=1}^{n} r_{i}x_{i} = m$. Hence, $x_{1},\ldots,x_{n}$ is a basis for $M$. Conversely, suppose $M$ is a finitely generated free $R$-module. Let $\{y_{1},y_{2},\ldots,y_{m}\}$ be a finitely generating set of $M$ and let $S$ be a basis of $M$. Then for each $1 \leq i \leq m$, we have $y_{i} = \sum_{j=1}^{N_{i}} r_{j}x_{ij}$ for some $N_{i} \in \N$, $r_{1},\ldots,r_{N_{i}} \in R$ and $x_{i1},\ldots,x_{iN_{i}} \in S$. So, $\{x_{ij} \mid 1 \leq i \leq m, \, 1 \leq j \leq N_{i}\}$ is finite and generates $M$. Hence $S$ is finite, so $S = \{x_{1},\ldots,x_{n}\}$ for some $n \in \N$. Define $\varphi : R^{n} \to M$ by $\varphi(r_{1},\ldots,r_{n}) = \sum_{i=1}^{n} r_{i}x_{i}$ for all $r_{1},\ldots,r_{n} \in R$. Then $\varphi$ is well-defined since if $(r_{1},\ldots,r_{n}) = (r_{1}',\ldots,r_{n}')$ for some $r_{1},\ldots,r_{n},r_{1}',\ldots,r_{n}' \in R$, then $r_{i} = r_{i}'$ for all $1 \leq i \leq n$, so $\sum_{i=1}^{n} r_{i}x_{i} = \sum_{i=1}^{n} r_{i}'x_{i}$. Also, for all $(r_{1},\ldots,r_{n}), (s_{1},\ldots,s_{n}) \in R^{n}$ and $t \in R$, $\varphi((r_{1},\ldots,r_{n}) + (s_{1},\ldots,s_{n})) = \varphi(r_{1}+s_{1},\ldots,r_{n}+s_{n}) = \sum_{i=1}^{n} (r_{i}+s_{i})x_{i} = \sum_{i=1}^{n} r_{i}x_{i} + \sum_{i=1}^{n} s_{i}x_{i} = \varphi(r_{1},\ldots,r_{n}) + \varphi(s_{1},\ldots,s_{n})$ and $\varphi(t(r_{1},\ldots,r_{n})) = \varphi(tr_{1},\ldots,tr_{n}) = t\sum_{i=1}^{n} r_{i}x_{i} = t\varphi(r_ {1},\ldots,r_ {n})$, so $\varphi$ is an $R$-linear map. Moreover, $\ker \varphi = 0$ since if $\varphi(r_{1},\ldots,r_{n}) = 0$ for some $r_{1},\ldots,r_{n} \in R$, then $\sum_{i=1}^{n} r_{i}x_{i} = 0$, so $r_{1}x_{1} = r_{2}x_{2} = \cdots = r_{n}x_{n} = 0$, so $r_{1} = r_{2} = \cdots = r_{n} = 0$. Also, $\Img \varphi = M$ since for all $m \in M$, there exist $r_{1},\ldots,r_{n} \in R$ such that $m = \sum_{i=1}^{n} r_{i}x_{i}$, so $\varphi(r_{1},\ldots,r_{n}) = m$. Hence, $\varphi$ is an $R$-module isomorphism, so $M \cong R^{n}$.
\end{proof}

\begin{proposition}
    Let $M$ and $N$ be $R$-modules. Then, giving an $R$-linear map $\varphi:M \to N$ is equivalent to giving an element of $N$ for each element of a basis of $M$. More precisely, if $M$ is a free $R$-module with basis $S$, then for any function $\bar{\varphi} : S \to N$, there exists a unique $R$-linear map $\varphi : M \to N$ such that $\varphi(s) = \bar{\varphi}(s)$ for all $s \in S$.
\end{proposition}

\begin{proof}
    Given an $R$-linear map, $\bar{\varphi} \defeq \varphi|_{S} : S \to N$ is a function. Conversely, given a function $\bar{\varphi} : S \to N$, define $\varphi : M \to N$ by $\varphi(\sum_{i=1}^{n} r_{i}s_{i}) = \sum_{i=1}^{n} r_{i}\bar{\varphi}(s_{i})$ for all $n \in \N$, $r_{1},\ldots,r_{n} \in R$ and $s_{1},\ldots,s_{n} \in S$. It is simple to check that $\varphi$ is well-defined and $R$-linear, and that $\varphi(s) = \bar{\varphi}(s)$ for all $s \in S$. Uniqueness also follows trivially.
\end{proof}

\begin{proposition}
    Let $M$ be an $R$-module and $I \subseteq R$ be an ideal. Then $IM = \{\sum_{i=1}^{n} x_{i}m_{i} \mid n \geq 1, \, x_{1},\ldots,x_{n} \in I, \, m_{1},\ldots,m_{n} \in M\}$ is an $R$-submodule of $M$. Moreover, $M/IM$ is an $R/I$-module via the operations defined by $(r+I)(m+IM) = rm + IM$ for all $r \in R$ and $m \in M$.
\end{proposition}

Before we being the proof, recall that via the quotient map $q:R \to R/I$, every module over $R/I$ can be regarded as a module over $R$ by defining $r \cdot m = q(r)m$ for all $r \in R$ and $m \in M$. The converse, however, is not true in general; that is, not every $R$-module can be regarded as an $R/I$-module.

\begin{proof}
    Note that $IM \subseteq M$ is clear. Let $\alpha_{1},\alpha_{2} \in IM$ and $r \in R$. Then $\alpha_{1} = x_{1}m_{1} + \cdots + x_{r}m_{r}$ and $\alpha_{2} = y_{1}n_{1} + \cdots + y_{s}n_{s}$ for some $r,s \in \N$, $x_{1},\ldots,x_{r},y_{1},\ldots,y_{s} \in I$ and $m_{1},\ldots,m_{r},n_{1},\ldots,n_{s} \in M$. Then $r\alpha_{1} + \alpha_{2} = r(x_{1}m_{1} + \cdots + x_{r}m_{r}) + (y_{1}n_{1} + \cdots + y_{s}n_{s}) = (rx_{1})m_{1} + \cdots + (rx_{r})m_{r} + y_{1}n_{1} + \cdots + y_{s}n_{s}$, so $r\alpha_{1} + \alpha_{2} \in IM$. Hence, $IM$ is an $R$-submodule of $M$. Thus, the quotient module $M/IM$ is well-defined, and an $R$-module under via $r \cdot (m + IM) = rm + IM$ for all $r \in R$ and $m \in M$. Moreover, for $r,r' \in R$, $m,m' \in M$ and $x \in I$, we have $(r+I)((r'+I)(m+IM)) = (r+I)(r'm + IM) = rr'm + IM$ and $((r+I)(r'+I))(m+IM) = (rr' + I)(m+IM) = rr'm + IM$, so the above definition indeed gives $M/IM$ the structure of an $R/I$-module.
\end{proof}

\begin{example}
    Note that $M$ being a finitely generated $R$-module does not imply that an $R$-submodule of $M$ is finitely generated. For instance, let $R = \Q[X_{1},X_{2},\ldots]$ be the polynomial ring in countably many variables over $\Q$, and let $M = R$. Then $M$ is a finitely generated $R$-module since $M = R \cdot 1$. However, the ideal $I = (X_{1},X_{2},\ldots)$ of $R$ is not finitely generated, so $I$ is an $R$-submodule of $M$ that is not finitely generated.
\end{example}

\subsection{Noetherian Modules}

\begin{proposition}
    Let $M$ be an $R$-module. Then the following are equivalent.
    \begin{enumerate}[label=\arabic*.]
        \item Every submodule of $M$ is finitely generated.
        \item Every ascending chain of submodules of $M$ stabilizes.
        \item Every non-empty set of submodules of $M$ has a maximal element under inclusion.
    \end{enumerate}
\end{proposition}

Such a module is called a \eax{noetherian module}.

\begin{proof}
    \begin{enumerate}[label=\arabic*.]
        \item Let $N = \bigcup_{i \geq 1} M_{i}$, for some ascending chain $M_{1} \subseteq M_{2} \subseteq \cdots$ of submodules of $M$. Then $N$ is a submodule of $M$, so $N$ is finitely generated by, say, $m_{1},\ldots,m_{k}$. Take $n \in \N$ such that $m_{1},\ldots,m_{k} \in M_{n}$. Then $M_{n} \subseteq M_{n+1} \subseteq \cdots \subseteq N$, so $M_{n} = M_{n+1} = \cdots = N$. Hence, the ascending chain stabilizes.
        \item Let $\Omega$ be a non-empty set of submodules of $M$. If $\Omega$ has no maximal element, then for all $N \in \Omega$, there exists $N' \in \Omega$ such that $N \subsetneq N'$. Hence, we can construct an infinite ascending chain $N_{1} \subsetneq N_{2} \subsetneq \cdots$ of submodules of $M$ by choosing $N_{1} \in \Omega$ arbitrarily and choosing $N_{i+1} \in \Omega$ such that $N_{i} \subsetneq N_{i+1}$ for all $i \geq 1$. This contradicts the assumption that every ascending chain of submodules of $M$ stabilizes. Hence, $\Omega$ has a maximal element.
        \item Let $N$ be a submodule of $M$. Let $\Omega = \{N' \subseteq N \mid N' \text{ is a submodule of } M\}$. Then $\Omega$ is non-empty since $0 \in \Omega$. Hence, $\Omega$ has a maximal element, say $N_{0}$. We claim that $N_{0} = N$. If not, then there exists $m \in N \setminus N_{0}$. Then $N_{0} + Rm$ is a submodule of $M$ containing $N_{0}$ properly, so $N_{0} + Rm \in \Omega$, contradicting the maximality of $N_{0}$. Hence, $N_{0} = N$, so $N$ is finitely generated by the maximal element of $\Omega$.
    \end{enumerate}
\end{proof}

\begin{proposition}
    Let $M$ be a finitely generated $R$-module, and let $N \subseteq M$ be a submodule. Then $M/N$ is finitely generated.
\end{proposition}

\begin{proof}
    Let $m_{1},\ldots,m_{k}$ be such that $M = Rm_{1} + \cdots + Rm_{k}$. For $\alpha = m + N \in M/N$, there exist $r_{1},\ldots,r_{k} \in R$ such that $m = r_{1}m_{1} + \cdots + r_{k}m_{k}$, so $\alpha = m + N = r_{1}m_{1} + \cdots + r_{k}m_{k} + N = r_{1}(m_{1}+N) + \cdots + r_{k}(m_{k}+N)$, so $M/N = R(m_{1}+N) + \cdots + R(m_{k}+N)$, so $M/N$ is finitely generated.
\end{proof}

\begin{proposition}
    Let $M$ be a noetherian $R$-module and $N \subseteq M$ be a submodule. Then $M/N$ is noetherian.
\end{proposition}

\begin{proof}
    Let $K \subseteq M/N$ be an $R$-submodule of $M/N$. Then $\tilde{K} = q^{-1}(K)$ is a submodule of $M$ containing $N$, where $q : M \to M/N$ is the quotient map. Since $M$ is noetherian, $\tilde{K}$ is finitely generated. By the previous proposition, $K = \tilde{K}/N$ is finitely generated. Hence, $M/N$ is noetherian. 
\end{proof}

\begin{theorem}
    Let $R$ be a noetherian ring. If $M$ is finitely generated as an $R$-module, then $M$ is a noetherian module.
\end{theorem}

\begin{proof}
    Let $x_{1},\ldots,x_{n} \in M$ be a generating set for $M$. Then $\varphi: R^{n} \to M$ extends to an $R$-linear sujrective map. By the first isomorphism theorem, $M \cong R^{n} / \ker \varphi$. Thus it is enough to prove that $R^{n}$ is noetherian, since the previous proposition implies that a quotient of a noetherian module is noetherian. We shall prove this by induction. For $n = 1$, it is trivially true since an $R$-submodule of $R$ is an ideal of $R$, and $R$ is noetherian so ideals of $R$ are finitely generated. Now assume $R^{n-1}$ is noetherian. Look at the inclusion map $i:R^{n-1} \to R^{n}$ defined by $i(r_{1},\ldots,r_{n-1}) = (r_{1},\ldots,r_{n-1},0)$ for all $r_{1},\ldots,r_{n-1} \in R$, and the projection map $\pi:R^{n} \to R$ defined by $\pi(r_{1},\ldots,r_{n}) = r_{n}$ for all $r_{1},\ldots,r_{n} \in R$. Then $i$ is an injective $R$-linear map, and $\pi$ is a surjective $R$-linear map. So, $\ker \pi \cong R^{n-1}$. Let $M_{1} = M \bigcap \ker \pi$, which is isomorphic to an $R$-submodule of $R^{n-1}$. Hence, $M_{1}$ is finitely generated, by the induciton hypothesis, by, say, $\{y_{1},\ldots,y_{r}\} \subseteq M_{1}$. Also, $\pi(M) \subseteq R$ is an $R$-submodule of $R$, that is, an ideal of $R$. So $\pi(M)$ is finitely generated by $\{z_{1},\ldots,z_{s}\} \subseteq \pi(M)$. Let $\tilde{z}_{i}$ be such that $\pi(\tilde{z}_{i}) = z_{i}$ for all $1 \leq i \leq s$. We claim that $\{y_{1},\ldots,y_{r},\tilde{z}_{1},\ldots,\tilde{z}_{s}\}$ generates $M$. Let $\alpha \in M$. Then $\pi(\alpha) = \sum_{i=1}^{s} r_{i}z_{i}$ for some $r_{1},\ldots,r_{s} \in R$, so $\pi(\alpha - \sum_{i=1}^{s} r_{i}\tilde{z}_{i}) = 0$, so $\alpha - \sum_{i=1}^{s} r_{i}\tilde{z}_{i} \in M \cap \ker \pi = M_{1}$, so $\alpha - \sum_{i=1}^{s} r_{i}\tilde{z}_{i} = \sum_{j=1}^{r} t_{j}y_{j}$ for some $t_{1},\ldots,t_{r} \in R$, so $\alpha = \sum_{j=1}^{r} t_{j}y_{j} + \sum_{i=1}^{s} r_{i}\tilde{z}_{i}$, so $\{y_{1},\ldots,y_{r},\tilde{z}_{1},\ldots,\tilde{z}_{s}\}$ generates $M$. Hence, $M$ is noetherian.
\end{proof}

\section{Localization}
\textit{March 30th.}

\begin{definition}
    Let $R$ be a ring and $S \subseteq R$ be a multiplicative subset, and let $M$ be an $R$-module. On $S \times M$, we define a relation $(s_{1},m_{1}) \sim (s_{2},m_{2})$ if there exists $t \in S$ such that $t(s_{2}m_{1} - s_{1}m_{2}) = 0$. It is simple to check that $\sim$ is an equivalence relation. Define $S^{-1}M \defeq S \times M/\sim$; the equivalence classes of $S^{-1}M$ are denoted by $\frac{m}{s}$ for $m \in M$ and $s \in S$. $S^{-1}M$ has a natural $S^{-1}R$-module structure given by $\frac{m}{s} + \frac{n}{t} = \frac{tm + sn}{st}$ and $\frac{r}{s} \cdot \frac{m}{t} = \frac{rm}{st}$ for all $m,n \in M$, $r \in R$ and $s,t \in S$. We call $S^{-1}M$ the localization of $M$ at $S$. These are well-defined and satisfy the axioms of an $S^{-1}R$-module. In particular, $S^{-1}M$ is also an $R$-module via $r \cdot \frac{m}{s} = \frac{rm}{s}$ for all $r \in R$, $m \in M$ and $s \in S$. Moreover, $\varphi:M \to S^{-1}M$ defined by $m \mapsto \frac{m}{1}$ for all $m \in M$ is an $R$-linear map.
\end{definition}

All of these are straightforward to verify and mimic the construction of the localization of a ring. 

\begin{example}
    One can simply take $M = R$, and $S^{-1}M$ is just the localization of $R$ at $S$, and works as an $S^{-1}R$-module under the natural operations.
\end{example}

\begin{example}
    Let $R = \Z$ and $M = \Z \oplus \Z/15\Z$. Taking $S = \Z \setminus \{0\}$, we have $S^{-1}M = S^{-1}\Z \oplus S^{-1}(\Z/15\Z) = \Q \oplus 0$; note that $S^{-1}(\Z/15\Z)$ is trivial since $15 \in S$ and $15m = 0$ for all $m \in \Z/15\Z$. Thus, $S^{-1}M = \Q$. Let us now take $S = \{1,3,9,27,\ldots\}$. Then
    \begin{align}
        S^{-1}M = S^{-1}\Z \oplus S^{-1}(\Z/3\Z) \oplus S^{-1}(\Z/5\Z) = \Z[\frac{1}{3}] \oplus 0 \oplus \Z/5\Z = \Z[\frac{1}{3}] \oplus \Z/5\Z. 
    \end{align}
    This is how we work in general; if $M$ is a direct sum of modules, then we can localize each component and take the direct sum of the localizations to get the localization of $M$. If $S = \{1,15,15^{2},15^{3},\ldots\}$, then show that $S^{-1}M = \Z[\frac{1}{15}]$.
\end{example}

In the above, we used some basic properties of localization without explicit proof. These include that $S^{-1}(M \oplus N) = S^{-1}M \oplus S^{-1}N$, $S^{-1}(M/N) = S^{-1}M/S^{-1}N$, and $S^{-1}(R/I) = S^{-1}R/S^{-1}I$ for an ideal $I$ of $R$. 

\textit{April 2nd.}

\begin{proposition}
    If $N \subseteq M$ is an $R$-submodule, then $S^{-1}N$ is an $R$-submodule of $S^{-1}M$, and $S^{-1}(M/N) \cong S^{-1}M/S^{-1}N$ as $S^{-1}R$-modules.
\end{proposition}
\begin{proof}
    Look at the map $\alpha:S^{-1}M \to S^{-1}(M/N)$ defined by $\frac{m}{s} \mapsto \frac{m+N}{s}$ for all $m \in M$ and $s \in S$. TO show this is well-defined, let $\frac{m}{s} = \frac{m'}{s'}$ for some $m,m' \in M$ and $s,s' \in S$. Then there exists $t \in S$ such that $t(s'm - sm') = 0$, which implies that $t(s'm+N - sm' - N) = N$ in $M/N$, so $\frac{m+N}{s} = \frac{m'+N}{s'}$. It is simple to check that $\alpha$ is an $S^{-1}R$-linear map, and is surjective. Moreover,
    \begin{align}
        \ker \alpha &= \{\frac{m}{s} \mid \frac{m+N}{s} = N\} = \{\frac{m}{s} \mid um \in N \text{ for some } u \in S\} = S^{-1}N
    \end{align}
    where last equality follows from the definition of $S^{-1}N$. Hence, by the first isomorphism theorem, $S^{-1}M/S^{-1}N \cong S^{-1}(M/N)$ as $S^{-1}R$-modules.
\end{proof}

\subsection{Rank}

\begin{definition}
    Let $R$ be an integral domain, and $M$ an $R$-module. The \eax{rank} of $M$, as an $R$-module, is defined as $\rank M = \dim_{S^{-1}R}(S^{-1}M)$, where $S = R \setminus \{0\}$. Note that $S^{-1}R$ is the field of fractions of $R$, so $S^{-1}M$ is a vector space over $S^{-1}R$, so $\dim_{S^{-1}R}(S^{-1}M)$ is well-defined.
\end{definition}

\begin{example}
    Let $R = \Z$ and $M = \Z^{2} \oplus \Q \oplus \Z/15\Z \oplus \Z/4\Z$. Then $S^{-1}M = \Q^{2} \oplus \Q \oplus 0 \oplus 0$, so $\rank M = 3$. If we let $M = n\Z$ instead, then $S^{-1}M = \Q$, so $\rank M = 1$. This is also easy to interpret since $n\Z$ is isomorphic to $\Z$ as a $\Z$-module, so $n\Z$ has the same rank as $\Z$, which is $1$.
\end{example}

\begin{example}
    Let $R = \Q[X,Y]$ and $I = (X,Y)$ be the ideal of $R$ generated by $X$ and $Y$. Then $S^{-1}I = S^{-1}R$, so $\rank I = 1$. 
\end{example}

We now come to the universal property of localization, which is a very important property that characterizes the localization of a module up to isomorphism.

\begin{theorem}[The \eax{universal property of localization}]
    Let $M$ be an $R$-module and $S \subseteq R$ be a multiplicative subset. Moreover, let $\varphi:M \to S^{-1}M$ be the $R$-linear localization map, $N$ be an $S^{-1}R$-module, and $\theta:M \to N$ be an $R$-linear map. Then there exists a unique $S^{-1}R$-linear map $\tilde{\theta}:S^{-1}M \to N$ such that $\tilde{\theta} \circ \varphi = \theta$.
\end{theorem}

\begin{proof}
    Define $\tilde{\theta}:S^{-1}M \to N$ by $\tilde{\theta}(\frac{m}{s}) = \frac{1}{s}\theta(m)$ for all $m \in M$ and $s \in S$. It may be verified that $\tilde{\theta}$ is well-defined and $S^{-1}R$-linear, and that $\tilde{\theta} \circ \varphi = \theta$. Uniqueness is also left as an exercise.
\end{proof}

\subsection{Torsion}

\begin{definition}
    Let $M$ be an $R$-module. Define
    \begin{align}
        T(M) = \{m \in M \mid rm = 0 \text{ for some } r \in R \setminus \{0\}\} \subseteq M.
    \end{align}
    $T(M)$ here is called the \eax{torsion} of $M$. Note that if $R$ is an integral domain, then $T(M)$ is an $R$-submodule of $M$.
\end{definition}
We show the second statement.

\begin{proof}
    Let $m_{1},m_{2} \in T(M)$. Then there exist $r_{1},r_{2} \in R \setminus \{0\}$ such that $r_{1}m_{1} = 0$ and $r_{2}m_{2} = 0$. Let $r = r_{1}r_{2}$, which is non-zero since $R$ is an integral domain. Then $r(m_{1}+m_{2}) = r_{1}(r_{2}m_{1}) + r_{2}(r_{1}m_{2}) = 0$, so $m_{1}+m_{2} \in T(M)$. Also, for $r' \in R$, we have $r'(rm_{1}) = (rr')m_{1} = 0$, so $r'm_{1} \in T(M)$. Hence, $T(M)$ is an $R$-submodule of $M$.
\end{proof}

\begin{example}
    Let $R = \Z/6\Z$, with $M = \Z/6\Z$. Then $T(M) = \{0,2,3,4\}$, which is not a submodule of $M$ since $2 + 3 = 5 \notin T(M)$.
\end{example}

\begin{definition}
    An $R$-module $M$ is called \eax{torsion free} if $T(M) = 0$. $M$ is called a \eax{torsion module} if $T(M) = M$.
\end{definition}

\begin{proposition}
    Suppose $R$ is an integral domain, and $M$ an $R$-module. Then $M/T(M)$ is torsion free. 
\end{proposition}


\textit{April 3rd.}

\begin{proof}
    Let $m + T(M) \in M/T(M)$ be such that $r(m + T(M)) = T(M)$ for some $r \in R \setminus \{0\}$. Then $rm \in T(M)$, so there exists $r' \in R \setminus \{0\}$ such that $r'(rm) = (rr')m = 0$, so $m \in T(M)$, so $m + T(M) = T(M)$. Hence, $M/T(M)$ is torsion free.
\end{proof}

\begin{proposition}
    Let $R$ be integral domain, and $M$ be a torsion $R$-module. Then $\rank M = 0$.
\end{proposition}

\begin{proof}
    Let $S = R \setminus \{0\}$. Then $S^{-1}M = 0$ since for all $\frac{m}{s} \in S^{-1}M$, we have $rm = 0$ for some $r \in R \setminus \{0\}$, so $\frac{m}{s} = \frac{rm}{rs} = 0$. Hence, $\rank M = \dim_{S^{-1}R}(S^{-1}M) = 0$.
\end{proof}

\begin{proposition}
    $\rank(M\oplus N) = \rank M + \rank N$ for all $R$-modules $M$ and $N$.
\end{proposition}
The proof is left as a straightforward exercise.

\begin{theorem}[The \eax{structure theorem for finitely generated modules over a PID}]
    Let $R$ be a principal ideal domain, and $M$ be a finitely generated $R$-module. Then
    \begin{align}
        M \cong R^{k} \oplus R/(a_{1}) \oplus R/(a_{2}) \oplus \cdots \oplus R/(a_{n})
    \end{align}
    for some $k,n \geq 0$ and $a_{1},\ldots,a_{n} \in R$ such that $a_{1} \mid a_{2} \mid \cdots \mid a_{n}$. Note that $k = 0$ if and only if $M$ is a torsion module, and $n = 0$ if and only if $M$ is torsion free. Moreover, the above decomposition is unique up to isomorphism.
\end{theorem}

Interesting corollaries can be drawn.

\begin{corollary}
    Let $R$ be a principal ideal domain, and $M$ be a finitely generated torsion free $R$-module. Then $M$ is a free $R$-module.
\end{corollary}

\begin{theorem}[The \eax{structure theorem for finitely generated abelian groups}]
    Let $G$ be a finitely generated abelian group. Then
    \begin{align}
        G \cong \Z^{k} \oplus \Z/n_{1}\Z \oplus \Z/n_{2}\Z \oplus \cdots \oplus \Z/n_{m}\Z
    \end{align}
    for some $k,m \geq 0$ and $n_{1},\ldots,n_{m} \in \N$ such that $n_{1} \mid n_{2} \mid \cdots \mid n_{m}$.
\end{theorem}
\begin{proof}
    This is a direct consequence of the previous theorem, since a finitely generated abelian group is just a finitely generated $\Z$-module, and $\Z$ is a principal ideal domain.
\end{proof}

\begin{example}
    Let $R = \Z$ and $M = \Q$. This is not finitely generated as a $\Z$-module. Note that $\rank \Q = 1$ since $S^{-1}\Q = \Q$ as a $\Q$-vector space, so $\dim_{\Q}(S^{-1}\Q) = 1$. However, $\Q$ is not singly generated; let $\frac{p}{q},\frac{r}{s} \in \Q$. Then $rq\frac{p}{q} -ps\frac{r}{s} = 0$, so any set of length $2$ is linearly dependent.
\end{example}

\begin{proposition}
    Let $R$ be a principal ideal domain, and $F$ be a free module of rank $n$. Also let $N$ be a submodule of $F$. Then $N$ is also free and of rank $m \leq n$. Moreover, there is a basis $\{x_{1},x_{2},\ldots,x_{n}\}$ of $F$ and elements $a_{1},a_{2},\ldots,a_{m} \in R$ such that $\{a_{1}x_{1},a_{2}x_{2},\ldots,a_{m}x_{m}\}$ is a basis of $N$ and $a_{1} \mid a_{2} \mid \cdots \mid a_{m}$.
\end{proposition}

\begin{proof}
    This is trivial if $N = 0$, so assume $N \neq 0$. Let $\Sigma = \{\varphi(N) \mid \varphi : F \to R \text{ is an } R\text{-linear map}\}$. Then $\Sigma$ is a non-empty collection of $R$-ideals, so $\Sigma$ has a maximal element, say $I_{1} = (a_{1}) = R$, with $a_{1} \neq 0$. Also, $(a_{1}) = \nu(N)$ for some $R$-linear $\nu : F \to R$. There exists $x \in N$ such that $a_{1} = \nu(x)$. We claim that $a_{1} \mid \varphi(x)$ for all $R$-linear maps $\varphi : F \to R$. If not, then let $d = \gcd (a_{1},\varphi(x))$ so that $d = r_{1}a_{1} + r_{2}\varphi(x)$ for some $r_{1},r_{2} \in R$. Consider $(r_{1}\nu + r_{2}\varphi) : F \to R$. Then $(r_{1}\nu + r_{2}\varphi)(x) = r_{1}a_{1} + r_{2}\varphi(x) = d$, which implies $(r_{1}\nu+r_{2}\varphi)(N) \supseteq (d) \supsetneq (a_{1})$, contradicting the maximality of $(a_{1})$.
    
    In particular, $a_{1} \mid \pi_{i} \circ \psi(x)$, where $\psi : F \to R^{n}$ is an isomorphism and $\pi_{i} : R^{n} \to R$ is the projection map defined by $\pi_{i}(r_{1},\ldots,r_{n}) = r_{i}$ for all $r_{1},\ldots,r_{n} \in R$. Let $\psi(x) = (b_{1},\ldots,b_{n})$. Then $a_{1} \mid b_{i}$ for all $1 \leq i \leq n$, so there exists $x_{1} \in F$ such that $\psi(x_{1}) = (\frac{b_{1}}{a_{1}}, \ldots, \frac{b_{n}}{a_{1}}) \in R^{n}$, and $a_{1}x_{1} = x$ (possible since $\psi(a_{1}x_{1}) = (b_{1},\ldots,b_{n})$). We now claim that $F = Rx_{1} \oplus \ker \nu$ and $N = Ra_{1}x_{1} \oplus (N \cap \ker \nu)$. To this end, let $y \in F$. Then $y = \nu(y)x_{1} + (y - \nu(y)x_{1})$, where $\nu(y)x_{1} \in Rx_{1}$ and $y - \nu(y)x_{1} \in \ker \nu$ (since $\nu(a_{1}x_{1}) = \nu(x)$, $\nu(x_{1}) = 1$). Now let $y \in Rx_{1} \cap \ker \nu$. Then $y = rx_{1}$ for some $r \in R$, and $0 = \nu(y) = r\nu(x_{1}) = r$, so $y = 0$. Hence, $F = Rx_{1} \oplus \ker \nu$. A similar argument shows that $N = Ra_{1}x_{1} \oplus (N \cap \ker \nu)$. Thus, $\rank N = \rank(Ra_{1}x_{1}) + \rank(N \cap \ker \nu)$ and $\rank F = \rank (Rx_{1}) + \rank(\ker \nu)$. Note that $\ker \nu \cap N$ is an $R$-submodule of $F$ and of rank strictly less than $\rank N$. Hence, by induction on the rank of $N$, $\ker \nu \cap N$ is free and of rank $\rank(N) - 1 = m-1$. But $N = Ra_{1}x_{1} \oplus (N \cap \ker \nu) \cong R^{m}$. Therefore, $N$ is free and of rank $m$.

    Now induct on the rank of $F$; $\ker \nu$ is an $R$-submodule of $F$ and of rank $n-1$. Also, it is free by what we just proved. Hence, $\ker \nu \cong R^{n-1}$. $N \cap \ker \nu \subseteq N$ is an $R$-submodule of $N$, so by the induction hypothesis, there exists a basis $\{x_{2},\ldots,x_{n}\}$ of $\ker \nu$ and elements $a_{2},\ldots,a_{m} \in R$ such that $a{2} \mid \cdots \mid a_{m}$ and $\{a_{2}x_{2},\ldots,a_{m}x_{m}\}$ is a basis of $N \cap \ker \nu$. By the claim above, $\{x_{1},\ldots,x_{n}\}$ is a basis of $F$ and $\{a_{1}x_{1},a_{2}x_{2},\ldots,a_{m}x_{m}\}$ is a basis of $N$. We now wish to show that $a_{1} \mid a_{2}$.

    With $\Sigma$ defined as beofre, and $(a_{1}) = \nu(N)$, we have $(a_{2}) = \nu_{2}(N \cap \ker \nu)$ for some $R$-linear $\nu_{2} : \ker \nu \to R$ and $\mu : Rx_{1} \oplus \ker \nu \to R$. For $r \in R$ and $\alpha \in \ker \nu$, we have $\mu(rx_{1}+\alpha) = r\nu(x_{1}) + \nu_{2}(\alpha)$. Hence, $\mu$ is $R$-linear and $\mu(x) = \mu(a_{1}x_{1}) = a_{1} \in \mu(N)$ so that $(a_{1}) = \mu(N)$. But $\mu(N) \supseteq \mu(\ker \nu \cap N) = \nu_{2}(\ker \nu \cap N) = (a_{2})$, so $a_{1} \mid a_{2}$.
\end{proof}
